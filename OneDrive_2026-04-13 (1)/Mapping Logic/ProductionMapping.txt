  WITH BASE_DOCS AS (SELECT 
    CLIENT,  
    PRODUCTIONORDER,  
    PLANT, 
    STORAGELOCATION, 
    BATCHNUMBER,  
    MATERIALNUMBER,  
    MOVEMENTCODE,  
    QUANTITY,  
    AMOUNTINLOCALCURRENCY,  
    LOCALCURRENCY, 
    VALUEOFTOTALVALUATEDSTOCK, 
     CONSUMPTIONQUANTITY,  
    REPLENISHMENTQUANTITY,  
    UNITOFMEASURE,  
    EVENTTIME,  
    MATERIALDOCUMENTNUMBER,  
    MATERIALDOCUMENTYEAR,  
    PLANTCOUNTRY,  
    DEBITCREDITINDICATOR, 
    AFKO.GLTRS AS SCHEDULEDFINISH, 
    AFKO.GSTRS AS SCHEDULEDSTART ,
    "Country of Origin" 
  FROM 
   <%=DATASOURCE:OCDM_SCHEMA%>."o_custom_MaterialDocumentItem" AS "o_custom_MaterialDocumentItem"
  LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>."AFKO" AS AFKO 
  ON AFKO.AUFNR = o_custom_MaterialDocumentItem.PRODUCTIONORDER  
  AND AFKO.MANDT = o_custom_MaterialDocumentItem.CLIENT 
  WHERE 
    1 = 1 
    AND PRODUCTIONORDER IS NOT  NULL
    AND MATERIALNUMBER IS NOT NULL 
    AND BATCHNUMBER IS NOT NULL 
    AND PLANT IS NOT NULL 
    AND STORAGELOCATION IS NOT NULL 
    AND ( 
      ( 
        MOVEMENTCODE IN ('261','262') 
      ) 
      OR ( 
        MOVEMENTCODE IN ( '101' ,'102')
      -- AND MOVEMENTCODETEXT = 'GR for order' 
      ) 
    ) 
  )
    ,GI AS ( SELECT 
        PRODUCTIONORDER, CLIENT, PLANT, STORAGELOCATION, BATCHNUMBER, MATERIALNUMBER, 
        --MOVEMENTCODE,
         UNITOFMEASURE, LOCALCURRENCY, 
        --MATERIALDOCUMENTNUMBER, MATERIALDOCUMENTYEAR, 
        PLANTCOUNTRY,"Country of Origin",
        MIN(EVENTTIME) AS EVENTTIME, 
        
        SUM(
            CASE
                WHEN MOVEMENTCODE = '261' THEN CONSUMPTIONQUANTITY
                WHEN MOVEMENTCODE = '262' THEN -CONSUMPTIONQUANTITY
                ELSE 0
            END
        ) AS  SUM_CONSUMPTIONQUANTITY,
        
        SUM(
            CASE
                WHEN MOVEMENTCODE = '261' THEN AMOUNTINLOCALCURRENCY
                WHEN MOVEMENTCODE = '262' THEN -AMOUNTINLOCALCURRENCY
                ELSE 0
            END
        ) AS SUM_AMOUNTINLOCALCURRENCY
        --SUM(CONSUMPTIONQUANTITY) AS SUM_CONSUMPTIONQUANTITY, 
        --SUM(AMOUNTINLOCALCURRENCY) AS SUM_AMOUNTINLOCALCURRENCY, 
        --COUNT(MATERIALDOCUMENTNUMBER) AS COUNT_DOCS,
        --MATERIALDOCUMENTNUMBER || ' - ' || MATERIALDOCUMENTYEAR AS MOVEMENTSOURCE
    FROM BASE_DOCS
    WHERE MOVEMENTCODE IN ('261','262') 
    GROUP BY 
        PRODUCTIONORDER, CLIENT, PLANT, STORAGELOCATION, BATCHNUMBER, MATERIALNUMBER, 
        UNITOFMEASURE, LOCALCURRENCY, 
        --MATERIALDOCUMENTNUMBER, MATERIALDOCUMENTYEAR, MOVEMENTCODE,
         PLANTCOUNTRY, "Country of Origin"),
 GR AS(
    SELECT 
        PRODUCTIONORDER, CLIENT, PLANT, STORAGELOCATION, BATCHNUMBER, MATERIALNUMBER, 
     UNITOFMEASURE, LOCALCURRENCY,-- MATERIALDOCUMENTNUMBER, 
        --MATERIALDOCUMENTYEAR, 
        PLANTCOUNTRY,"Country of Origin",
        SUM(
            CASE
                WHEN MOVEMENTCODE = '101' THEN REPLENISHMENTQUANTITY
                WHEN MOVEMENTCODE = '102' THEN -REPLENISHMENTQUANTITY
                ELSE 0
            END
        ) AS  SUM_REPLENISHMENTQUANTITY,
    
        SUM(
            CASE
                WHEN MOVEMENTCODE = '101' THEN AMOUNTINLOCALCURRENCY
                WHEN MOVEMENTCODE = '102' THEN -AMOUNTINLOCALCURRENCY
                ELSE 0
            END
        ) AS SUM_AMOUNTINLOCALCURRENCY,

        --SUM(REPLENISHMENTQUANTITY) AS SUM_REPLENISHMENTQUANTITY, 
        --SUM(AMOUNTINLOCALCURRENCY) AS SUM_AMOUNTINLOCALCURRENCY, 
        --COUNT(MATERIALDOCUMENTNUMBER) AS COUNT_DOCS, 
        MAX(EVENTTIME) AS EVENTTIME
        --MATERIALDOCUMENTNUMBER || ' - ' || MATERIALDOCUMENTYEAR AS MOVEMENTTARGET
    FROM BASE_DOCS
    WHERE MOVEMENTCODE IN ('101' ,'102')
    GROUP BY 
        PRODUCTIONORDER, CLIENT, PLANT, STORAGELOCATION, BATCHNUMBER, MATERIALNUMBER, 
        UNITOFMEASURE, LOCALCURRENCY, --MATERIALDOCUMENTNUMBER, MATERIALDOCUMENTYEAR, 
         
        PLANTCOUNTRY,"Country of Origin"
),
COO_FLAGS AS (
    SELECT
        BATCHNUMBER,

        MAX(
          CASE WHEN "Country of Origin" = 'DE' THEN 1 ELSE 0 END
        ) AS HAS_DE,

        MAX(
          CASE WHEN "Country of Origin" = 'NL' THEN 1 ELSE 0 END
        ) AS HAS_NL

    FROM BASE_DOCS   
    GROUP BY BATCHNUMBER
)
SELECT DISTINCT
    'ProductionMapping_'||GI.CLIENT||GI.MATERIALNUMBER||GI.PLANT|| GI.BATCHNUMBER || GI.STORAGELOCATION || --GI.MATERIALDOCUMENTNUMBER || 
    GR.CLIENT||GR.MATERIALNUMBER||GR.PLANT ||GR.BATCHNUMBER ||GR.STORAGELOCATION || GR.PRODUCTIONORDER     --|| GR.MATERIALDOCUMENTNUMBER
     AS ID, 
     GI.CLIENT AS CLIENTSOURCE, 
    GI.EVENTTIME AS EVENTTIMESOURCE, 
    GI.MATERIALNUMBER AS MATERIALNUMBERSOURCE, 
    GI.PLANT AS PLANTSOURCE, 
    GI.STORAGELOCATION AS STORAGELOCATIONSOURCE, 
    GI.BATCHNUMBER AS BATCHNUMBERSOURCE, 
    --GI.MOVEMENTSOURCE, 
    GI.SUM_CONSUMPTIONQUANTITY AS QUANTITYSOURCE, 
    GI.UNITOFMEASURE AS QUANTITYSOURCEUOM, 
    GI.SUM_AMOUNTINLOCALCURRENCY AS LOCALVALUATIONSOURCE, 
    GI.LOCALCURRENCY AS LOCALCURRENCYSOURCE, 
    --GI.MOVEMENTCODE AS MOVEMENTTYPESOURCE, 
    --GI.COUNT_DOCS AS MATERIALDOCUMENTCOUNTSOURCE, 
CASE
  WHEN HAS_DE = 1 AND HAS_NL = 1 THEN 'NL,DE'
  WHEN HAS_NL  = 1 THEN 'NL'
  WHEN HAS_DE = 1 THEN 'DE'
  ELSE '' END
AS "Country of Origin",

    GI.PLANTCOUNTRY AS COUNTRYSOURCE,  
    GR.CLIENT AS CLIENTTARGET, 
    GR.EVENTTIME AS EVENTTIMETARGET, 
    GR.MATERIALNUMBER AS MATERIALNUMBERTARGET, 
    GR.PLANT AS PLANTTARGET, 
    GR.STORAGELOCATION AS STORAGELOCATIONTARGET, 
    GR.BATCHNUMBER AS BATCHNUMBERTARGET, 
    GR.PRODUCTIONORDER AS PRODUCTIONNUMBER, 
    GR.SUM_REPLENISHMENTQUANTITY AS QUANTITYTARGET, 
    GR.UNITOFMEASURE AS QUANTITYTARGETUOM, 
    GR.SUM_AMOUNTINLOCALCURRENCY AS LOCALVALUATIONTARGET, 
    GR.LOCALCURRENCY AS LOCALCURRENCYTARGET, 
    GR.PLANTCOUNTRY AS COUNTRYTARGET,  
    'Production Mapping' AS LINKTYPE, 
    'MaterialMasterPlant_' || GI.CLIENT || GI.MATERIALNUMBER || GI.PLANT AS MaterialMasterPlant,
    'NodeTSN_' || GI.CLIENT || GI.MATERIALNUMBER || GI.PLANT || GI.STORAGELOCATION || GI.BATCHNUMBER AS "Node" 
FROM GI 
INNER JOIN GR 
    ON GI.PRODUCTIONORDER = GR.PRODUCTIONORDER 
    AND GI.CLIENT = GR.CLIENT 
    --AND GI.PLANT = GR.PLANT
LEFT JOIN COO_FLAGS COO
  ON COO.BATCHNUMBER = GI.BATCHNUMBER
