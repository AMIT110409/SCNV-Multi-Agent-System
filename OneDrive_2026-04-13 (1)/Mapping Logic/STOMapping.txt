WITH QUANTITY_MAPPING AS (

  SELECT 
    CLIENT, 
    MATERIALDOCUMENTYEAR,
     MATERIALNUMBER, 
    PLANT, 
    STORAGELOCATION, 
    BATCHNUMBER, 
    SUM(QUANTITY) AS SUM_QUANTITY, 
    SUM(AMOUNTINLOCALCURRENCY) AS SUM_AMOUNTINLOCALCURRENCY, 
    LOCALCURRENCY AS LOCALCURRENCY, 
    -- SUM(CONSUMPTIONQUANTITY) AS SUM_CONSUMPTIONQUANTITY, 
    --SUM(REPLENISHMENTQUANTITY) AS SUM_REPLENISHMENTQUANTITY, 
    SUM(CASE 
    WHEN MOVEMENTCODE IN ('641','643') AND DEBITCREDITINDICATOR = 'H' THEN CONSUMPTIONQUANTITY
    WHEN MOVEMENTCODE = '644' AND DEBITCREDITINDICATOR = 'S' THEN -ABS(CONSUMPTIONQUANTITY)
    ELSE 0 END) AS 
    SUM_CONSUMPTIONQUANTITY,
SUM(CASE 
    WHEN MOVEMENTCODE = '101' AND DEBITCREDITINDICATOR = 'S' THEN REPLENISHMENTQUANTITY
    WHEN MOVEMENTCODE = '102' AND DEBITCREDITINDICATOR = 'H' THEN -ABS(REPLENISHMENTQUANTITY)
    ELSE 0 END) AS 
    SUM_REPLENISHMENTQUANTITY,
    UNITOFMEASURE, 
    PLANTCOUNTRY AS SOURCEPLANTCOUNTRY, 
    PURCHASEORDER, 
    PURCHASEORDERITEM
  FROM 
 <%=DATASOURCE:OCDM_SCHEMA%>."o_custom_MaterialDocumentItem"  AS MaterialDocumentItem 
WHERE 
1 = 1 
AND MATERIALNUMBER IS NOT NULL 
AND (PLANT IS NOT NULL OR RECEIVINGPLANT IS NOT NULL) 
AND STORAGELOCATION IS NOT NULL  
AND BATCHNUMBER IS NOT NULL 
AND MOVEMENTCODE IN ('641', '643','644','102','101') 
--AND PURCHASEORDER IS NOT NULL
GROUP BY 
CLIENT, 
MATERIALDOCUMENTYEAR,
MATERIALNUMBER, 
PLANT, 
STORAGELOCATION, 
BATCHNUMBER,   
UNITOFMEASURE, 
LOCALCURRENCY, 
PLANTCOUNTRY, 
PURCHASEORDER, 
PURCHASEORDERITEM
)

, TSN_STO_DOMESTIC AS 

 ( 
  SELECT 
    MaterialDocumentItem.CLIENT, 
    MaterialDocumentItem.MATERIALDOCUMENTNUMBER, 
    MaterialDocumentItem.MATERIALDOCUMENTYEAR,
     MaterialDocumentItem.MATERIALNUMBER, 
    MaterialDocumentItem.PLANT, 
    MaterialDocumentItem.STORAGELOCATION, 
    MaterialDocumentItem.BATCHNUMBER, 
    MaterialDocumentItem.MOVEMENTCODE, 
    MaterialDocumentItem.DEBITCREDITINDICATOR, 
    MaterialDocumentItem.EVENTTIME, 
    QM.SUM_QUANTITY, 
    QM.SUM_AMOUNTINLOCALCURRENCY, 
    MaterialDocumentItem.LOCALCURRENCY AS LOCALCURRENCY, 
    QM.SUM_CONSUMPTIONQUANTITY, 
    QM.SUM_REPLENISHMENTQUANTITY, 
    MaterialDocumentItem.UNITOFMEASURE, 
    MaterialDocumentItem.PLANTCOUNTRY AS SOURCEPLANTCOUNTRY, 
    MaterialDocumentItem.PURCHASEORDER, 
    MaterialDocumentItem.PURCHASEORDERITEM,
    MaterialDocumentItem.RECEIVINGPLANT, 
    MaterialDocumentItem.RECEIVINGSTORAGELOCATION,
    T001W.LAND1 AS RECEIVINGCOUNTRY  
  FROM 
 <%=DATASOURCE:OCDM_SCHEMA%>."o_custom_MaterialDocumentItem"  AS MaterialDocumentItem 
 JOIN QUANTITY_MAPPING AS QM ON 1=1
 AND QM.CLIENT                    =  MaterialDocumentItem.CLIENT
AND QM.MATERIALDOCUMENTYEAR       = MaterialDocumentItem.MATERIALDOCUMENTYEAR
AND QM.MATERIALNUMBER            =  MaterialDocumentItem.MATERIALNUMBER
AND QM.PLANT                     =  MaterialDocumentItem.PLANT
AND QM.STORAGELOCATION           =  MaterialDocumentItem.STORAGELOCATION
AND QM.BATCHNUMBER              =   MaterialDocumentItem.BATCHNUMBER
AND QM.PURCHASEORDER             =  MaterialDocumentItem.PURCHASEORDER
AND QM.PURCHASEORDERITEM          = MaterialDocumentItem.PURCHASEORDERITEM
--WHERE SUM_CONSUMPTIONQUANTITY <> 0 AND SUM_REPLENISHMENTQUANTITY <> 0
LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.T001W ON 1=1 
    AND T001W.MANDT = MaterialDocumentItem.CLIENT  
    AND T001W.WERKS = MaterialDocumentItem.PLANT 
LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKET AS EKET 
  ON EKET.MANDT = MaterialDocumentItem.CLIENT 
 AND EKET.EBELN = MaterialDocumentItem.PURCHASEORDER 
 AND EKET.EBELP = MaterialDocumentItem.PURCHASEORDERITEM 
INNER  JOIN ( SELECT  
    MANDT,  
    EBELN,  
    EBELP,  
    MAX(ETENR) AS MAX_ETENR 
FROM  
<%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKET 
GROUP BY  
MANDT, EBELN, EBELP) M  
ON EKET.MANDT = M.MANDT  
AND EKET.EBELN = M.EBELN  
AND EKET.EBELP = M.EBELP  
AND EKET.ETENR = M.MAX_ETENR 
LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKKO AS EKKO ON 
1=1 
AND EKKO.MANDT = MaterialDocumentItem.CLIENT 
AND EKKO.EBELN = MaterialDocumentItem.PURCHASEORDER 
WHERE 
1 = 1 
AND MaterialDocumentItem.MATERIALNUMBER IS NOT NULL 
AND (MaterialDocumentItem.PLANT IS NOT NULL OR MaterialDocumentItem.RECEIVINGPLANT IS NOT NULL) 
AND MaterialDocumentItem.STORAGELOCATION IS NOT NULL  
AND MaterialDocumentItem.BATCHNUMBER IS NOT NULL


)

--|| GOODS_ISSUE.MATERIALDOCUMENTNUMBER || GOODS_ISSUE.MATERIALDOCUMENTITEM
||GOODS_RECEIPT.CLIENT||GOODS_RECEIPT.MATERIALNUMBER||GOODS_RECEIPT.PLANT||GOODS_RECEIPT.STORAGELOCATION || GOODS_RECEIPT.BATCHNUMBER || 
 GOODS_ISSUE.PURCHASEORDER|| GOODS_ISSUE.PURCHASEORDERITEM ||GOODS_ISSUE.MATERIALDOCUMENTNUMBER || ' - ' || GOODS_ISSUE.MATERIALDOCUMENTYEAR 
 || GOODS_RECEIPT.MATERIALDOCUMENTNUMBER || ' - ' ||GOODS_RECEIPT.MATERIALDOCUMENTYEAR 
 AS ID,

    GOODS_ISSUE.LOCALCURRENCY AS LOCALCURRENCYSOURCE,

    GOODS_ISSUE.SOURCEPLANTCOUNTRY AS COUNTRYSOURCE, 
    GOODS_RECEIPT.CLIENT AS CLIENTTARGET, 
    -- GOODS_RECEIPT.EVENTTIME AS EVENTTIMETARGET, 
    GOODS_RECEIPT.MATERIALNUMBER AS MATERIALNUMBERTARGET, 
    GOODS_RECEIPT.PLANT AS PLANTTARGET, 
    GOODS_RECEIPT.STORAGELOCATION AS STORAGELOCATIONTARGET, 
    GOODS_RECEIPT.BATCHNUMBER AS BATCHNUMBERTARGET, 
    --GOODS_RECEIPT.MATERIALDOCUMENTNUMBER || ' - ' ||GOODS_RECEIPT.MATERIALDOCUMENTYEAR ,
    --AS MOVEMENTTARGET, 
    NULL AS PRODUCTIONNUMBER, 
    NULL AS SALESDOCUMENT, 
    GOODS_ISSUE.PURCHASEORDER AS PURCHASEDOCUMENT, 
    GOODS_ISSUE.PURCHASEORDERITEM AS PURCHASEORDERITEM,
    GOODS_RECEIPT.PURCHASEORDER AS PURCHASEDOCUMENTX, 
    GOODS_RECEIPT.PURCHASEORDERITEM AS PURCHASEORDERITEMX,
    GOODS_RECEIPT.SUM_REPLENISHMENTQUANTITY AS QUANTITYTARGET, 
    GOODS_RECEIPT.UNITOFMEASURE AS QUANTITYTARGETUOM,

    GOODS_RECEIPT.LOCALCURRENCY AS LOCALCURRENCYTARGET, 
    GOODS_RECEIPT.MOVEMENTCODE AS MOVEMENTTYPETARGET,

    'Stock Transferred from Lochem' AS LINKTYPE, 
    
'MaterialMasterPlant_'||GOODS_ISSUE.CLIENT||GOODS_ISSUE.MATERIALNUMBER||GOODS_ISSUE.PLANT AS MaterialMasterPlant, 
    
'MaterialMasterPlant_'||GOODS_ISSUE.CLIENT||GOODS_ISSUE.MATERIALNUMBER||GOODS_ISSUE.PLANT AS MaterialMasterPlantTSN, 
    'NodeTSN_' || GOODS_ISSUE.CLIENT || GOODS_ISSUE.MATERIALNUMBER || 
GOODS_ISSUE.PLANT || GOODS_ISSUE.STORAGELOCATION || GOODS_ISSUE.BATCHNUMBER AS "Node" ,
    ROW_NUMBER() OVER (
      PARTITION BY
        GOODS_ISSUE.CLIENT,
        GOODS_ISSUE.MATERIALNUMBER,GOODS_RECEIPT.MATERIALNUMBER ,
        GOODS_ISSUE.PLANT,GOODS_RECEIPT.PLANT,
        GOODS_ISSUE.STORAGELOCATION,
        GOODS_ISSUE.RECEIVINGPLANT,
        GOODS_ISSUE.BATCHNUMBER,GOODS_RECEIPT.BATCHNUMBER ,
        GOODS_ISSUE.PURCHASEORDER,
        GOODS_ISSUE.PURCHASEORDERITEM
      ORDER BY GOODS_ISSUE.EVENTTIME
    ) AS RN
  FROM 
    "TSN_STO_DOMESTIC" AS GOODS_ISSUE 
    JOIN "TSN_STO_DOMESTIC" AS GOODS_RECEIPT ON 1 = 1 
    AND GOODS_ISSUE.CLIENT = GOODS_RECEIPT.CLIENT 
    AND GOODS_ISSUE.MATERIALNUMBER = GOODS_RECEIPT.MATERIALNUMBER 
    AND GOODS_ISSUE.RECEIVINGPLANT = GOODS_RECEIPT.PLANT 
    --AND GOODS_ISSUE.RECEIVINGSTORAGELOCATION = GOODS_RECEIPT.STORAGELOCATION 
    AND GOODS_ISSUE.BATCHNUMBER = GOODS_RECEIPT.BATCHNUMBER 
    AND GOODS_ISSUE.PURCHASEORDER = GOODS_RECEIPT.PURCHASEORDER 
    AND GOODS_ISSUE.PURCHASEORDERITEM = GOODS_RECEIPT.PURCHASEORDERITEM 
    AND GOODS_ISSUE.MATERIALDOCUMENTYEAR = GOODS_RECEIPT.MATERIALDOCUMENTYEAR 
   -- group by 1 order by 2 desc
 
--WHERE  GOODS_ISSUE.BATCHNUMBER = '10GBXZF'
)
