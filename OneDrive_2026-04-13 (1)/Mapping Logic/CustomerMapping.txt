
WITH TSN_CUSTOMER_ORDERS AS (
    SELECT
        -- Aggregates
        COALESCE(MAX(VBAK.KUNNR), ' ') AS KUNNR, 
        COALESCE(MAX(KNA1.NAME1), ' ') AS KUNNR_TEXT, 
        COALESCE(MAX(KNA1.LAND1), ' ') AS KUNNR_LAND1, 
        --COUNT(MDI.MATERIALDOCUMENTNUMBER) AS COUNT_MATERIALDOCUMENTNUMBER, 
        SUM(MDI.AMOUNTINLOCALCURRENCY) AS SUM_AMOUNTINLOCALCURRENCY, 
        SUM(MDI.CONSUMPTIONQUANTITY) AS SUM_CONSUMPTIONQUANTITY, 
        SUM(MDI.REPLENISHMENTQUANTITY) AS SUM_REPLENISHMENTQUANTITY, 
        
        -- Dimensions (Moved to Group By)
        KNA1.WERKS AS CustomerPlant, 
        CASE 
            WHEN KNA1.MANDT IS NOT NULL THEN
                CASE 
                    WHEN KNA1.VBUND IS NULL THEN 'External' 
                    ELSE 'Internal' 
                END 
            ELSE 'No Customer found in Master Data' 
        END AS KUNNR_EXTERN_INTERN, 
        MDI.LOCALCURRENCY, 
        MDI.DELIVERYNUMBER AS DeliveryHeader, 
        MDI.DELIVERYITEMNUMBER AS DeliveryItem, 
        VBAP.VBELN AS SalesHeader,  
        VBAP.POSNR AS SalesItem, 
        VBAP.NETWR AS NetOrderValue, 
        VBAP.WAERK AS SOCurrency, 
        VBAP.VRKME AS QuantityUnit,                                                                                                                 
        VBAP.KWMENG AS OrderedQuantity, 
        VBAK.VKORG AS SalesOrganization, 
        TVKOT.VTEXT AS SalesOrganizationText, 
        MDI.CLIENT,  
        MDI.MATERIALDOCUMENTNUMBER,  
        MDI.MATERIALDOCUMENTYEAR,  
        MDI.MATERIALNUMBER,  
        MDI.PLANT, 
        MDI.STORAGELOCATION, 
        MDI.BATCHNUMBER, 
        MDI.MOVEMENTCODE,  
        MDI.MOVEMENTCODETEXT,  
        MDI.EVENTTIME,  
        MDI.UNITOFMEASURE, 
        MDI.PLANTCOUNTRY
        -- VBAK.AUDAT AS PLANNEDSTART, 
        -- VBAK.VDATU AS PLANNEDEND, 
        -- VTTK.DATEN AS ACTUALEND,
        -- -- Calculated lead times can stay in CTE or move to Outer Select
        -- DATEDIFF(DAY, VBAK.AUDAT, VBAK.VDATU) AS PLANNED_LEADTIME_DAYS, 
        -- DATEDIFF(DAY, VBAK.AUDAT, VTTK.DATEN) AS ACTUAL_LEADTIME_DAYS 

    FROM <%=DATASOURCE:OCDM_SCHEMA%>."t_o_custom_MaterialDocumentItem" AS MDI
    INNER JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.LIPS AS LIPS
        ON MDI.CLIENT = LIPS.MANDT
        AND MDI.BATCHNUMBER = LIPS.CHARG
        AND MDI.PLANT = LIPS.WERKS
        AND MDI.DELIVERYNUMBER = LIPS.VBELN
    INNER JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.VBAP AS VBAP
        ON LIPS.MANDT = VBAP.MANDT
        AND LIPS.VGBEL = VBAP.VBELN
        AND LIPS.VGPOS = VBAP.POSNR
    INNER JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.VBAK AS VBAK
        ON VBAK.MANDT = VBAP.MANDT
        AND VBAK.VBELN = VBAP.VBELN
    LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.KNA1 AS KNA1
        ON KNA1.MANDT = VBAK.MANDT
        AND KNA1.KUNNR = VBAK.KUNNR
    LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.VTTP AS VTTP
        ON VTTP.MANDT = LIPS.MANDT
        AND VTTP.VBELN = LIPS.VBELN
    LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.VTTK AS VTTK
        ON VTTK.MANDT = VTTP.MANDT
        AND VTTK.TKNUM = VTTP.TKNUM
    LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.TVKOT AS TVKOT
        ON TVKOT.MANDT = VBAK.MANDT
        AND TVKOT.VKORG = VBAK.VKORG
        AND TVKOT.SPRAS = 'E'
WHERE 
    1 = 1 
    AND MATERIALNUMBER IS NOT NULL 
    AND PLANT IS NOT NULL 
    AND BATCHNUMBER IS NOT NULL 
    AND STORAGELOCATION IS NOT NULL 
    AND MOVEMENTCODE IN ('601') 
    AND DELIVERYNUMBER IS NOT NULL 
    GROUP BY
        MDI.CLIENT, MDI.MATERIALDOCUMENTNUMBER, MDI.MATERIALDOCUMENTYEAR, MDI.MATERIALNUMBER, 
        MDI.PLANT, MDI.STORAGELOCATION, MDI.BATCHNUMBER, MDI.MOVEMENTCODE, MDI.MOVEMENTCODETEXT, 
        MDI.EVENTTIME, MDI.UNITOFMEASURE, MDI.LOCALCURRENCY, MDI.DELIVERYNUMBER, MDI.DELIVERYITEMNUMBER,
        MDI.PLANTCOUNTRY, VBAP.VBELN, VBAP.POSNR, VBAP.NETWR, VBAP.WAERK, VBAP.VRKME, VBAP.KWMENG,
        VBAK.AUDAT, VBAK.VDATU, VBAK.VKORG, KNA1.MANDT, KNA1.VBUND, KNA1.LAND1, KNA1.WERKS, 
        VTTK.DATEN, TVKOT.VTEXT
)
SELECT DISTINCT 
    'CustomerMapping_' || CONFIG.CLIENT || CONFIG.MATERIALNUMBER || CONFIG.PLANT || CONFIG.STORAGELOCATION || CONFIG.BATCHNUMBER || CONFIG.MATERIALDOCUMENTNUMBER ||
    CONFIG.CLIENT || CONFIG.MATERIALNUMBER || CONFIG.DeliveryHeader || CONFIG.DeliveryItem 
    || CONFIG.SalesHeader || CONFIG.SalesItem||  'CUST - ' || CONFIG.KUNNR AS ID,
    CONFIG.CLIENT AS CLIENTSOURCE, 
    CONFIG.EVENTTIME AS EVENTTIMESOURCE, 
    CONFIG.MATERIALNUMBER AS MATERIALNUMBERSOURCE, 
    CONFIG.PLANT AS PLANTSOURCE, 
    CONFIG.STORAGELOCATION AS STORAGELOCATIONSOURCE, 
    CONFIG.MATERIALDOCUMENTNUMBER || ' - ' || CONFIG.MATERIALDOCUMENTYEAR AS MOVEMENTSOURCE, 
    CONFIG.BATCHNUMBER AS BATCHSOURCE, 
    CONFIG.SUM_CONSUMPTIONQUANTITY AS QUANTITYSOURCE, 
    CONFIG.UNITOFMEASURE AS QUANTITYSOURCEUOM, 
    CONFIG.SUM_AMOUNTINLOCALCURRENCY AS AMOUNTLOCALCURRENCYSOURCE, 
    CONFIG.LOCALCURRENCY AS LOCALCURRENCYSOURCE, 
    CONFIG.MOVEMENTCODE AS MOVEMENTTYPESOURCE, 
    --CONFIG.COUNT_MATERIALDOCUMENTNUMBER AS MATERIALDOCUMENTCOUNTSOURCE, 
    CONFIG.KUNNR AS CUSTOMERNUMBER, 
    CONFIG.PLANTCOUNTRY AS COUNTRYSOURCE,
    CONFIG.KUNNR_TEXT AS CUSTOMERNAME, 
    CONFIG.KUNNR AS VENDORNUMBER, 
    CONFIG.CLIENT AS CLIENTTARGET, 
    CONFIG.EVENTTIME AS EVENTTIMETARGET, 
    CONFIG.MATERIALNUMBER AS MATERIALNUMBERTARGET, 
    COALESCE(CustomerPlant, '/CUST - ' || CONFIG.KUNNR) AS PLANTTARGET, 
    CONFIG.STORAGELOCATION AS STORAGELOCATIONTARGET, 
    CONFIG.MATERIALDOCUMENTNUMBER || ' - ' || CONFIG.MATERIALDOCUMENTYEAR AS MOVEMENTTARGET, 
    CONFIG.BATCHNUMBER AS BATCHTARGET, 
    NULL AS PRODUCTIONNUMBER, 
    CONFIG.SalesHeader AS SALESDOCUMENT, 
    CONFIG.SalesHeader AS SALESORDER, 
    CONFIG.SalesItem AS SALESORDERITEM, 
    CONFIG.DeliveryHeader AS DELIVERYNUMBER, 
    CONFIG.DeliveryItem AS DELIVERYITEMNUMBER, 
    NetOrderValue AS NETORDERITEMVALUE, 
    SOCurrency AS CURRENCY, 
    QuantityUnit AS QUANTITYUNIT,                                                                                                                   
    OrderedQuantity AS ORDEREDQUANTITY, 
    NULL AS PURCHASEDOCUMENT, 
    CONFIG.SUM_REPLENISHMENTQUANTITY AS QUANTITYTARGET,         
    CONFIG.UNITOFMEASURE AS QUANTITYTARGETUOM, 
    CONFIG.SUM_AMOUNTINLOCALCURRENCY AS AMOUNTLOCALCURRENCYTARGET, 
    CONFIG.LOCALCURRENCY AS LOCALCURRENCYTARGET, 
    CONFIG.MOVEMENTCODE AS MOVEMENTTYPETARGET, 
    --CONFIG.COUNT_MATERIALDOCUMENTNUMBER AS MATERIALDOCUMENTCOUNTTARGET, 
    CONFIG.KUNNR_LAND1 AS COUNTRYTARGET, 
    CONFIG.SalesOrganization AS SALESORGANIZATION, 
    CONFIG.SalesOrganizationText AS SALESORGANIZATIONTEXT, 
    CONFIG.KUNNR_EXTERN_INTERN AS CUSTOMERTYPEFLAG,  
    -- CONFIG.PLANNED_LEADTIME_DAYS AS PLANNEDLEADTIMEDAYS, 
    'Goods Issued To Customer' AS "LINKTYPE", 
    'MaterialMasterPlant_' || CONFIG.CLIENT || CONFIG.MATERIALNUMBER || CONFIG.PLANT AS MaterialMasterPlant,
    'NodeTSN_' || CONFIG.CLIENT || CONFIG.MATERIALNUMBER || CONFIG.PLANT || CONFIG.STORAGELOCATION || CONFIG.BATCHNUMBER AS "Node" 
FROM "TSN_CUSTOMER_ORDERS" AS CONFIG
 

-- CONFIG.CLIENT||CONFIG.MATERIALNUMBER|| CONFIG.DeliveryHeader || CONFIG.DeliveryItem || 'CUST - '||CONFIG.KUNNR
 WHERE  CONFIG.PLANT NOT IN ('3779','3973','3690','3736','3101','3159','3969','3158','3528','3423','3111','3778','3432','3013','3234','3110')
