WITH 
 "TSN_VENDOR_ISSUING" AS ( 
SELECT DISTINCT 
COALESCE(MAX(EKKO.LIFNR), ' ') AS LIFNR, 
COALESCE(MAX(LFA1.NAME1), ' ') AS LIFNR_TEXT, 
COALESCE(MAX(LFA1.LAND1),' ') AS LIFNR_LAND1, 
LFA1.WERKS AS PLANTVENDOR, 
CLIENT, 
MATERIALDOCUMENTNUMBER, 
MATERIALDOCUMENTYEAR, 
MATERIALDOCUMENTITEM,
MATERIALNUMBER, 
PLANT, 
BATCHNUMBER, 
PLANTCOUNTRY, 
STORAGELOCATION, 
MOVEMENTCODE, 
MOVEMENTCODETEXT, 
EVENTTIME, 
"Country of Origin",
-- COUNT(MATERIALDOCUMENTNUMBER) AS 
-- COUNT_MATERIALDOCUMENTNUMBER, 
SUM(AMOUNTINLOCALCURRENCY) AS SUM_AMOUNTINLOCALCURRENCY, 
LOCALCURRENCY AS LOCALCURRENCY, 
SUM(CONSUMPTIONQUANTITY) AS SUM_CONSUMPTIONQUANTITY, 
SUM(REPLENISHMENTQUANTITY) AS SUM_REPLENISHMENTQUANTITY, 
UNITOFMEASURE, 
PURCHASEORDER, 
PURCHASEORDERITEM, 
DELIVERYDATE
-- EKKO.BEDAT AS PLANNEDSTART, 
-- EKET.EINDT AS PLANNEDEND, 
-- EKKO.BEDAT AS ACTUALSTART, 
-- EVENTTIME AS ACTUALEND, 
-- DATEDIFF(DAY, EKKO.BEDAT, EKET.EINDT) AS PLANNED_LEADTIME_DAYS, 
-- DATEDIFF(DAY, EKKO.BEDAT, EVENTTIME) AS ACTUAL_LEADTIME_DAYS 
FROM 
<%=DATASOURCE:OCDM_SCHEMA%>."o_custom_MaterialDocumentItem" AS MaterialDocumentItem 
 LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKET AS EKET 
  ON EKET.MANDT = MaterialDocumentItem.CLIENT 
  AND EKET.EBELN = MaterialDocumentItem.PURCHASEORDER 
  AND EKET.EBELP = MaterialDocumentItem.PURCHASEORDERITEM 
 INNER  JOIN ( SELECT  
        MANDT,  
        EBELN,  
        EBELP,  
        MAX(ETENR) AS MAX_ETENR 
  FROM  
        <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKET 
    GROUP BY  
        MANDT, EBELN, EBELP) M  
    ON EKET.MANDT = M.MANDT  
    AND EKET.EBELN = M.EBELN  
    AND EKET.EBELP = M.EBELP  
    AND EKET.ETENR = M.MAX_ETENR 
INNER JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.EKKO AS EKKO ON 
1=1 
AND EKKO.MANDT = MaterialDocumentItem.CLIENT 
AND EKKO.EBELN = MaterialDocumentItem.PURCHASEORDER 
LEFT JOIN <%=DATASOURCE:DATA_INGESTION_POOL_-_PRODUCTION_-_SAP_(PLEASE_DON'T_TOUCH)_SAP_PE1%>.LFA1 AS LFA1 ON 
1=1 
AND LFA1.MANDT = EKKO.MANDT 
AND LFA1.LIFNR = EKKO.LIFNR 
WHERE 
1 = 1 
AND MATERIALNUMBER IS NOT NULL 
AND PLANT IS NOT NULL 
AND STORAGELOCATION IS NOT NULL 
AND BATCHNUMBER IS NOT NULL 
AND MOVEMENTCODE IN ('101') 
AND PURCHASEORDER IS NOT NULL
AND  LFA1.WERKS IS NULL
AND  NOT EXISTS (
    SELECT 1
    FROM <%=DATASOURCE:OCDM_SCHEMA%>."o_custom_MaterialDocumentItem" R
    WHERE R.CLIENT = MaterialDocumentItem.CLIENT
      AND R.PURCHASEORDER = MaterialDocumentItem.PURCHASEORDER
      AND R.PURCHASEORDERITEM = MaterialDocumentItem.PURCHASEORDERITEM
      AND R.BATCHNUMBER = MaterialDocumentItem.BATCHNUMBER
      AND R.MOVEMENTCODE = '102'
      -- Match on quantity to ensure it's a true reversal
      AND R.QUANTITY = MaterialDocumentItem.QUANTITY 
      AND R.EVENTTIME >= MaterialDocumentItem.EVENTTIME
      AND R.MATERIALDOCUMENTNUMBER != MaterialDocumentItem.MATERIALDOCUMENTNUMBER
)

GROUP BY 
CLIENT, 
MATERIALDOCUMENTNUMBER, 
MATERIALDOCUMENTYEAR, 
MATERIALNUMBER, 
PLANT, 
BATCHNUMBER, 
STORAGELOCATION, 
MOVEMENTCODE, 
MOVEMENTCODETEXT, 
UNITOFMEASURE, 
LOCALCURRENCY, 
EVENTTIME, 
DELIVERYDATE, 
PURCHASEORDER, 
PURCHASEORDERITEM, 
PLANTCOUNTRY, 
LFA1.LAND1, 
EKKO.BEDAT, 
EKET.EINDT, 
LFA1.WERKS ,
EKKO.LIFNR,
MATERIALDOCUMENTITEM,
"Country of Origin"
) ,
COO_FLAGS AS (
    SELECT
        BATCHNUMBER,

        MAX(
          CASE WHEN "Country of Origin" = 'DE' THEN 1 ELSE 0 END
        ) AS HAS_DE,

        MAX(
          CASE WHEN "Country of Origin" = 'NL' THEN 1 ELSE 0 END
        ) AS HAS_NL

    FROM TSN_VENDOR_ISSUING   -- this is your GI+GR result
    GROUP BY BATCHNUMBER)
SELECT DISTINCT 
'GOODSISSUED_' || GOODS_ISSUE.CLIENT||GOODS_ISSUE.MATERIALNUMBER||'/VEND - '||GOODS_ISSUE.LIFNR||GOODS_ISSUE.CLIENT||GOODS_ISSUE.MATERIALNUMBER||GOODS_ISSUE.PLANT||GOODS_ISSUE.STORAGELOCATION
||GOODS_ISSUE.BATCHNUMBER || GOODS_ISSUE.MATERIALDOCUMENTNUMBER ||GOODS_ISSUE.MATERIALDOCUMENTITEM  AS ID, --changed 
GOODS_ISSUE.CLIENT AS CLIENTSOURCE,
GOODS_ISSUE.DELIVERYDATE AS EVENTTIMESOURCE, 
GOODS_ISSUE.MATERIALNUMBER AS MATERIALNUMBERSOURCE, 
COALESCE('/VEND - '|| GOODS_ISSUE.PLANTVENDOR,'/VEND - '||GOODS_ISSUE.LIFNR) AS PLANTSOURCE, 
GOODS_ISSUE.MATERIALDOCUMENTNUMBER || ' - ' || 
GOODS_ISSUE.MATERIALDOCUMENTYEAR AS MOVEMENTSOURCE, 
GOODS_ISSUE.BATCHNUMBER AS BATCHSOURCE, --changed 
GOODS_ISSUE.STORAGELOCATION AS STORAGELOCATIONSOURCE, 
GOODS_ISSUE.SUM_CONSUMPTIONQUANTITY AS QUANTITYSOURCE, 
GOODS_ISSUE.LOCALCURRENCY AS LOCALCURRENCYSOURCE, 
GOODS_ISSUE.UNITOFMEASURE AS QUANTITYSOURCEUOM, 
GOODS_ISSUE.SUM_AMOUNTINLOCALCURRENCY AS 
LOCALVALUATIONSOURCE, 
GOODS_ISSUE.MOVEMENTCODE AS MOVEMENTTYPESOURCE, 
-- GOODS_ISSUE.COUNT_MATERIALDOCUMENTNUMBER AS 
-- MATERIALDOCUMENTCOUNTSOURCE, 
GOODS_ISSUE.LIFNR_LAND1 AS COUNTRYSOURCE, 
GOODS_ISSUE.LIFNR_TEXT AS VENDORNAME, 
GOODS_ISSUE.LIFNR AS VENDORNUMBER, 
GOODS_ISSUE.CLIENT AS CLIENTTARGET, 
GOODS_ISSUE.EVENTTIME AS EVENTTIMETARGET, 
GOODS_ISSUE.MATERIALNUMBER AS MATERIALNUMBERTARGET, 
GOODS_ISSUE.PLANT AS PLANTTARGET, 
GOODS_ISSUE.STORAGELOCATION AS STORAGELOCATIONTARGET, 
GOODS_ISSUE.MATERIALDOCUMENTNUMBER || ' - ' || 
GOODS_ISSUE.MATERIALDOCUMENTYEAR AS MOVEMENTTARGET, 
GOODS_ISSUE.BATCHNUMBER AS BATCHTARGET, --changed 
NULL AS PRODUCTIONNUMBER, 
NULL AS SALESDOCUMENT, 
GOODS_ISSUE.PURCHASEORDER AS PURCHASEDOCUMENT, 
GOODS_ISSUE.SUM_REPLENISHMENTQUANTITY AS QUANTITYTARGET, 
GOODS_ISSUE.UNITOFMEASURE AS QUANTITYTARGETUOM, 
GOODS_ISSUE.SUM_AMOUNTINLOCALCURRENCY AS 
LOCALVALUATIONTARGET, 
GOODS_ISSUE.LOCALCURRENCY AS LOCALCURRENCYTARGET, 
GOODS_ISSUE.MOVEMENTCODE AS MOVEMENTTYPETARGET, 
-- GOODS_ISSUE.COUNT_MATERIALDOCUMENTNUMBER AS 
-- MATERIALDOCUMENTCOUNTTARGET, 
GOODS_ISSUE.PLANTCOUNTRY AS COUNTRYTARGET, 
-- CAST(GOODS_ISSUE.PLANNEDSTART AS TIMESTAMP) AS PLANNEDSTART, 
-- CAST(GOODS_ISSUE.PLANNEDEND AS TIMESTAMP) AS PLANNEDEND, 
-- CAST(GOODS_ISSUE.ACTUALSTART AS TIMESTAMP) AS ACTUALSTART, 
-- CAST(GOODS_ISSUE.ACTUALEND AS TIMESTAMP) AS ACTUALEND, 
-- PLANNED_LEADTIME_DAYS AS PLANNEDLEADTIMEDAYS, 
-- ACTUAL_LEADTIME_DAYS AS ACTUALLEADTIMEDAYS, 
'GR From Vendor' AS LINKTYPE, 
'MaterialMasterPlant_'||GOODS_ISSUE.CLIENT||GOODS_ISSUE.MATERIALNUMBER||GOODS_ISSUE.PLANT AS MaterialMasterPlant, 
'NodeTSN_' || GOODS_ISSUE.CLIENT || GOODS_ISSUE.MATERIALNUMBER || 
GOODS_ISSUE.PLANT || GOODS_ISSUE.STORAGELOCATION || 
GOODS_ISSUE.BATCHNUMBER AS "Node" ,
CASE
  WHEN HAS_DE = 1 AND HAS_NL = 1 THEN 'NL,DE'
  WHEN HAS_NL  = 1 THEN 'NL'
  WHEN HAS_DE = 1 THEN 'DE'
  ELSE NULL END
AS "Country of Origin"
FROM "TSN_VENDOR_ISSUING" AS GOODS_ISSUE 
LEFT JOIN COO_FLAGS COO
  ON COO.BATCHNUMBER = GOODS_ISSUE.BATCHNUMBER

-- ||GOODS_ISSUE.BATCHNUMBER || GOODS_ISSUE.MATERIALDOCUMENTNUMBER  ||GOODS_ISSUE.MATERIALDOCUMENTITEM
